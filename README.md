## Install

```bash
npm install --save react-finderselect
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-finderselect'
import 'react-finderselect/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
